#!/usr/bin/perl 
# localhost:4747
use CGI qw{param}; генератор динамической HTML-страницы, param() - приём параметров, посланных в запросе пользователя
print "Content-type: text/html\n\n"; Без этой строки данные не будут выводиться в окно браузера
sub x { объявление функции x по умолчанию аргументы передаются в массив $_
  $y = $_[0]; $_[0] элемент массива _ записываем в переменную y
  print `echo $y 2>&1`; echo $y выводим значение y | 2>&1 означает перенаправление stderr в stdout
}
x(param("x")); вызов функции x с параметром пользователя

В Perl существует несколько операторов ввода-вывода. Первый это скобки из символа '`' - акцента. Строка в этих скобках воспринимается как системная команда и результат ее действия возвращается как "псевдо" литерал. Статус выполненной команды хранится в переменной $? .
Источник: http://perl.find-info.ru/perl/001/operator.htm#13
